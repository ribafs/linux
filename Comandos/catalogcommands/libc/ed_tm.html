tm

Descrição

Estrutura usada para armazenar data e hora.

Cabeçalho

Para usar esta estrutura de dados deve-se incluir o cabeçalho time.h no programa.

#include <time.h>

Formato

struct tm
{
    int tm_sec; /* segundos [0-60] */
    int tm_min; /* minutos [0-59] */
    int tm_hour; /* horas [0-23] */
    int tm_mday; /* dias [1-31] */
    int tm_mon; /* mes [0-11] */
    int tm_year; /* ano: 0 = 1900; 1 = 1901; ... */
    int tm_wday; /* dia da semana [0 = domingo; 1 = segunda; ...; 6 = sabado] */
    int tm_yday; /* dia do ano [0-365] */
    int tm_isdst; /* horario de verao (0 = nao, 1 = sim) */
};
